#######Change IP Address
sudo openssl req -x509 -nodes -days 3650 -newkey rsa:2048 \
-keyout /etc/ssl/private/netbox.key \
-out /etc/ssl/certs/netbox.crt \
-subj "/CN=<IP ADDRESS>" \
-addext "subjectAltName = IP:<IP ADDRESS>"
